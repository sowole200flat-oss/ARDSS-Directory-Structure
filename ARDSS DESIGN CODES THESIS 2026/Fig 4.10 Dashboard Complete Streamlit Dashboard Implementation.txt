# dashboard/app.py
import streamlit as st
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import json, os, sys

sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))

from modules.preprocessor import PreprocessingPipeline
from modules.analytics_engine import AnalyticsEngine
from modules.llm_layer import LLMLayer
from modules.risk_register import build_baseline_register, build_ai_assisted_register
from config import RISK_CATEGORIES, DATA_RAW_DIR

st.set_page_config(page_title="ARDSS — Reflective Risk Analytics",
                   layout="wide")

@st.cache_data
def load_and_process(use_llm: bool):
    """Cached pipeline run — avoids reprocessing on every user interaction."""
    with open(os.path.join(DATA_RAW_DIR, "retrospective_entries.json")) as f:
        raw_entries = json.load(f)
    pipeline = PreprocessingPipeline()
    processed = pipeline.process_corpus(raw_entries)
    engine = AnalyticsEngine()
    analytics_df = engine.run_full_pipeline(processed)
    if use_llm:
        llm = LLMLayer()
        llm_results = llm.classify_corpus(analytics_df)
        register = build_ai_assisted_register(analytics_df, llm_results)
    else:
        register = build_baseline_register(analytics_df)
    return register, engine

# ── Sidebar controls
st.sidebar.title("ARDSS Controls")
view = st.sidebar.radio("Select View",
    ["Overview", "Cluster Explorer", "Entry Inspector", "Sentiment Timeline"])
use_llm = st.sidebar.toggle("Enable AI-Assisted Mode (GPT-4o)", value=False)

if use_llm and "OPENAI_API_KEY" not in os.environ:
    st.sidebar.warning("Set OPENAI_API_KEY in your environment to use AI mode.")
    use_llm = False

register_df, engine = load_and_process(use_llm)
mode_label = "AI-Assisted" if use_llm else "Baseline (TF-IDF Only)"
st.title(f"ARDSS — Retrospective Risk Analytics Dashboard ({mode_label})")
st.caption("Outputs are analytical hypotheses for practitioner reflection —",
           "not authoritative risk assessments.")

# ── View: Overview
if view == "Overview":
    st.subheader("Risk Category Distribution")
    cat_col = "final_category" if use_llm else "risk_category"
    counts = register_df[cat_col].value_counts().reindex(RISK_CATEGORIES, fill_value=0)
    fig, ax = plt.subplots(figsize=(8, 4))
    sns.barplot(x=counts.index, y=counts.values, palette="Blues_d", ax=ax)
    ax.set_xlabel("Risk Category"); ax.set_ylabel("Number of Entries")
    ax.set_title("Entries per Risk Category")
    st.pyplot(fig)
    st.subheader("Risk Register Summary")
    display_cols = ["entry_id", "sprint", cat_col, "sentiment_label"]
    if use_llm:
        display_cols += ["confidence", "category_agreement"]
    st.dataframe(register_df[display_cols], use_container_width=True)

# ── View: Cluster Explorer
elif view == "Cluster Explorer":
    st.subheader("Cluster Keyword Explorer")
    keywords = engine.get_top_keywords_per_cluster()
    mapping = engine.cluster_to_category
    for cluster_idx, kws in keywords.items():
        category = mapping.get(cluster_idx, f"Cluster {cluster_idx}")
        with st.expander(f"Cluster {cluster_idx} → {category.upper()}"):
            st.write("Top keywords:", ", ".join(kws))
            cluster_entries = register_df[register_df["cluster_index"] == cluster_idx]
            st.write(f"{len(cluster_entries)} entries assigned to this cluster.")

# ── View: Entry Inspector
elif view == "Entry Inspector":
    st.subheader("Individual Entry Inspection")
    entry_ids = register_df["entry_id"].tolist()
    selected_id = st.selectbox("Select Entry", entry_ids)
    row = register_df[register_df["entry_id"] == selected_id].iloc[0]
    st.markdown("**Original Text**")
    st.info(row["text"])
    cat_col = "final_category" if use_llm else "risk_category"
    st.metric("Risk Category", row[cat_col])
    st.metric("Sentiment", row["sentiment_label"])
    st.metric("Sentiment Score", row["sentiment_compound"])
    st.markdown("**Cluster Keywords**")
    st.write(", ".join(row["cluster_keywords"]))
    if use_llm:
        st.markdown("**LLM Reasoning**")
        st.write(row.get("reasoning", "N/A"))
        st.markdown("**Supporting Evidence (LLM Excerpts)**")
        for ev in row.get("evidence", []):
            st.markdown(f"> {ev}")
        st.metric("Confidence", row["confidence"])
        st.metric("TF-IDF / LLM Agreement",
                  "Yes" if row["category_agreement"] else "No")

# ── View: Sentiment Timeline
elif view == "Sentiment Timeline":
    st.subheader("Sentiment Trend Across Sprint Sequence")
    timeline = register_df[["entry_id","sprint","sentiment_compound"]].copy()
    timeline = timeline.sort_values("entry_id")
    fig, ax = plt.subplots(figsize=(12, 4))
    colors = timeline["sentiment_compound"].map(
        lambda s: "#e74c3c" if s < -0.05 else ("#2ecc71" if s > 0.05 else "#95a5a6"))
    ax.bar(timeline["entry_id"], timeline["sentiment_compound"], color=list(colors))
    ax.axhline(0, color="black", linewidth=0.8, linestyle="--")
    ax.axhline(-0.05, color="#e74c3c", linewidth=0.5, linestyle=":",
               label="Negative threshold")
    ax.set_xlabel("Entry ID"); ax.set_ylabel("Compound Sentiment Score")
    ax.set_title("VADER Compound Sentiment by Retrospective Entry")
    ax.tick_params(axis="x", rotation=90); ax.legend()
    st.pyplot(fig)
    sprint_avg = timeline.groupby("sprint")["sentiment_compound"].mean()
    st.subheader("Average Sentiment by Sprint")
    st.dataframe(sprint_avg.rename("Mean Compound Score").round(4))
