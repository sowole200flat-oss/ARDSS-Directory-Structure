# tests/test_integration.py
import unittest, json, pandas as pd, os, sys
sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))
from modules.preprocessor import PreprocessingPipeline
from modules.analytics_engine import AnalyticsEngine
from modules.risk_register import build_baseline_register
from config import RISK_CATEGORIES

class TestFullPipelineIntegration(unittest.TestCase):
    @classmethod
    def setUpClass(cls):
        cls.sample_entries = [
            {"entry_id": f"TEST_{i}", "sprint": 1,
             "text": f"Sample entry {i} with risk signals about delays and blockers."}
            for i in range(5)
        ]
        pipeline = PreprocessingPipeline()
        cls.processed = pipeline.process_corpus(cls.sample_entries)
        engine = AnalyticsEngine()
        cls.analytics_df = engine.run_full_pipeline(cls.processed)
        cls.register = build_baseline_register(cls.analytics_df)

    def test_output_row_count(self):
        self.assertEqual(len(self.register), 5)

    def test_required_columns_present(self):
        required = ["entry_id", "sprint", "text", "risk_category",
                    "cluster_keywords", "sentiment_label",
                    "sentiment_compound", "condition"]
        for col in required:
            self.assertIn(col, self.register.columns, msg=f"Missing column: {col}")

    def test_risk_category_values_valid(self):
        valid = set(RISK_CATEGORIES + ["unassigned"])
        for cat in self.register["risk_category"]:
            self.assertIn(cat, valid)

    def test_sentiment_compound_range(self):
        for score in self.register["sentiment_compound"]:
            self.assertGreaterEqual(score, -1.0)
            self.assertLessEqual(score, 1.0)

    def test_sentiment_label_values_valid(self):
        valid = {"positive", "negative", "neutral"}
        for label in self.register["sentiment_label"]:
            self.assertIn(label, valid)

    def test_cluster_keywords_non_empty(self):
        for kws in self.register["cluster_keywords"]:
            self.assertIsInstance(kws, list)
            self.assertGreater(len(kws), 0)

if __name__ == "__main__":
    unittest.main()
