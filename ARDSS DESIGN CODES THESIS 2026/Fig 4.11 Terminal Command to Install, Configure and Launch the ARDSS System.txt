# 1. Install all pinned dependencies
pip install -r requirements.txt

# 2. Download required NLTK data packages (first run only)
# punkt_tab is required for NLTK 3.8.1+ — do not use punkt
python -c "import nltk; nltk.download(['punkt_tab', 'stopwords', 'wordnet', 'vader_lexicon'])"

# 3. Set OpenAI API key in environment (Linux/macOS)
# On Windows: set OPENAI_API_KEY=your-api-key-here
export OPENAI_API_KEY="your-api-key-here"

# 4. Launch the interactive dashboard
# Outputs: dashboard loads risk register from memory
# LLM logs written to: logs/llm_calls.log
# Evaluation CSVs written to: data/processed/
streamlit run dashboard/app.py
